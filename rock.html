// variable declaration let wins = 0, losses = 0, ties = 0; const options = ['r', 'p', 's']; function getRandomNumber(n) { return Math.floor(Math.random() * n); } function playGame(userSelection, cpuSelection) { let result = 0; // lets assume a tie if (userSelection === 'r') { // switch statement // can substitute if statements switch (cpuSelection) { case 's': result = 1; break; case 'p': result = -1; break; } } else if (userSelection == 's') { switch (cpuSelection) { case 'p': result = 1; break; case 'r': result = -1; break; } } else { switch (cpuSelection) { case 'r': result = 1; break; case 's': result = -1; break; } } return result; } function updateDOM(result, userSelection, cpuSelection) { const winsDOM = document.querySelector('#wins'); const lossesDOM = document.querySelector('#losses'); const tiesDOM = document.querySelector('#ties'); const totalDOM = document.querySelector('#total'); const resultDOM = document.querySelector('#result'); const userDOM = document.querySelector('#user-selection'); const cpuDOM = document.querySelector('#cpu-selection'); // Update user and cpu selections userDOM.textContent = userSelection.toUpperCase(); cpuDOM.textContent = cpuSelection.toUpperCase(); // Update the result if (result === 1) { resultDOM.textContent = 'You won!'; } else if (result === -1) { resultDOM.textContent = 'Looser!'; } else { resultDOM.textContent = 'Tie! Boring!!'; } // update wins, losses, ties, total winsDOM.textContent = wins; lossesDOM.textContent = losses; tiesDOM.textContent = ties; totalDOM.textContent = wins + losses + ties; } document.onkeyup = function(e) { // (find) is an array method :-) // (option => e.key === option) is an arrow function :-O // Look info of those const userSelection = e.key.toLowerCase(); if (options.find(option => userSelection === option)) { const n = getRandomNumber(options.length); // get a random number const cpuSelection = options[n]; const result = playGame(userSelection, cpuSelection); // lets play the game if (result === 1) { wins++; } else if (result == -1) { losses++; } else { ties++; } updateDOM(result, userSelection, cpuSelection); } };
